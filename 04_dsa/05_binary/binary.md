### Binary to base 10
0b1010 -> 10
0b0011 -> 3

### Binary to hexadecimal
0b1010 -> 0xA
0b0011 -> 0x3

### Hexadecimal to base 10
0xa1 -> 161
0xff -> 255

### Hexadecimal to binary
0xa1 -> 161 -> 0b10100001
0xff -> 255 -> 0b11111111

### Base 10 to binary
8 -> 0b100
24 -> 0b11000
255 -> 0b11111111

### Base 10 to hexadecimal
8 -> 0x8
24 -> 0x18
255 -> 0xff

### Base 10 to ASCII
65 -> A
66 -> B
97 -> a
98 -> b

### Hexadecimal to ASCII
0x41 -> A
0x42 -> B
0x61 -> a
0x62 -> b

### Binary to ASCII
0b01000001 -> 65 -> A
0b01000010 -> 66 -> B
0b01100001 -> 97 -> a
0b01100010 -> 98 -> b
